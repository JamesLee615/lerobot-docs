1. 概述

lerobot_train 使用离线数据集训练模仿学习策略（如 ACT、Diffusion Policy 等）。它通过 HuggingFace Accelerate 实现多 GPU 分布式训练，支持检查点保存/恢复，并可选择在仿真环境中定期评估。

3. train() 主函数流程

文件: src/lerobot/scripts/lerobot_train.py:154-546

@parser.wrap()
def train(cfg: TrainPipelineConfig, accelerator: Accelerator | None = None):
    # ======== 阶段 1: 配置验证 ========
    cfg.validate()

    # ======== 阶段 2: 创建 Accelerator ========
    if accelerator is None:
        ddp_kwargs = DistributedDataParallelKwargs(find_unused_parameters=True)
        accelerator = Accelerator(
            step_scheduler_with_optimizer=False,  # 手动控制 scheduler 步进
            kwargs_handlers=[ddp_kwargs],
            cpu=cfg.policy.device == "cpu",
        )

    init_logging(accelerator=accelerator)
    is_main_process = accelerator.is_main_process

    # ======== 阶段 3: WandB 初始化 ========
    if cfg.wandb.enable and is_main_process:
        wandb_logger = WandBLogger(cfg)

    # ======== 阶段 4: 随机种子 ========
    if cfg.seed is not None:
        set_seed(cfg.seed, accelerator=accelerator)

    # ======== 阶段 5: 设备与性能配置 ========
    device = accelerator.device
    torch.backends.cudnn.benchmark = True
    torch.backends.cuda.matmul.allow_tf32 = True

    # ======== 阶段 6: 创建数据集 ========
    # 主进程先加载 (避免多进程竞争下载)
    if is_main_process:
        dataset = make_dataset(cfg)
    accelerator.wait_for_everyone()
    if not is_main_process:
        dataset = make_dataset(cfg)

    # ======== 阶段 7: 创建评估环境 (可选) ========
    eval_env = None
    if cfg.eval_freq > 0 and cfg.env is not None and is_main_process:
        eval_env = make_env(cfg.env, n_envs=cfg.eval.batch_size)

    # ======== 阶段 8: 创建策略 ========
    policy = make_policy(
        cfg=cfg.policy,
        ds_meta=dataset.meta,     # 用于推断 input/output features
        rename_map=cfg.rename_map,
    )

    # ======== 阶段 8b: PEFT 包装 (可选) ========
    if cfg.peft is not None:
        policy = policy.wrap_with_peft(peft_cli_overrides=dataclasses.asdict(cfg.peft))

    accelerator.wait_for_everyone()

    # ======== 阶段 9: 创建处理器 ========
    preprocessor, postprocessor = make_pre_post_processors(
        policy_cfg=cfg.policy,
        pretrained_path=cfg.policy.pretrained_path,
        dataset_stats=dataset.meta.stats,
        ...
    )

    # ======== 阶段 10: 创建优化器和调度器 ========
    optimizer, lr_scheduler = make_optimizer_and_scheduler(cfg, policy)

    # ======== 阶段 10b: RA-BC 权重提供者 (可选) ========
    rabc_weights = None
    if cfg.use_rabc:
        rabc_weights = RABCWeights(
            progress_path=cfg.rabc_progress_path,
            chunk_size=policy.config.chunk_size,
            ...
        )

    # ======== 阶段 11: 恢复检查点 (可选) ========
    step = 0
    if cfg.resume:
        step, optimizer, lr_scheduler = load_training_state(
            cfg.checkpoint_path, optimizer, lr_scheduler
        )

    # ======== 阶段 12: 创建 DataLoader ========
    if hasattr(cfg.policy, "drop_n_last_frames"):
        # DP 训练: 使用 EpisodeAwareSampler
        sampler = EpisodeAwareSampler(
            dataset.meta.episodes["dataset_from_index"],
            dataset.meta.episodes["dataset_to_index"],
            drop_n_last_frames=cfg.policy.drop_n_last_frames,
            shuffle=True,
        )
        shuffle = False  # sampler 自己处理随机性
    else:
        shuffle = True
        sampler = None

    dataloader = DataLoader(
        dataset,
        num_workers=cfg.num_workers,
        batch_size=cfg.batch_size,
        shuffle=shuffle,
        sampler=sampler,
        pin_memory=(device.type == "cuda"),
        prefetch_factor=2,
    )

    # ======== 阶段 13: Accelerator.prepare ========
    policy, optimizer, dataloader, lr_scheduler = accelerator.prepare(
        policy, optimizer, dataloader, lr_scheduler
    )
    dl_iter = cycle(dataloader)  # 循环迭代器

    # ======== 阶段 14: 训练循环 ========
    for _ in range(step, cfg.steps):
        start_time = time.perf_counter()

        # 获取 batch + 预处理
        batch = next(dl_iter)
        batch = preprocessor(batch)
        # preprocessor: RenameObs → ToBatch → Device → Normalize

        # 单步更新
        train_tracker, output_dict = update_policy(
            train_tracker, policy, batch, optimizer,
            cfg.optimizer.grad_clip_norm,
            accelerator=accelerator,
            lr_scheduler=lr_scheduler,
            rabc_weights_provider=rabc_weights,
        )

        step += 1

        # 日志
        if step % cfg.log_freq == 0:
            logging.info(train_tracker)
            wandb_logger.log_dict(...)

        # 检查点
        if cfg.save_checkpoint and step % cfg.save_freq == 0:
            save_checkpoint(checkpoint_dir, step, cfg, policy, ...)
            update_last_checkpoint(checkpoint_dir)

        # 评估
        if cfg.env and step % cfg.eval_freq == 0:
            eval_info = eval_policy_all(eval_env, policy, ...)

    # ======== 阶段 15: 推送模型到 Hub ========
    if is_main_process and cfg.policy.push_to_hub:
        unwrapped_policy.push_model_to_hub(cfg)

    accelerator.end_training()

4. 训练循环：update_policy()

文件: src/lerobot/scripts/lerobot_train.py:59-150

def update_policy(train_metrics, policy, batch, optimizer, grad_clip_norm,
                  accelerator, lr_scheduler, lock, rabc_weights_provider):
    start_time = time.perf_counter()
    policy.train()

    # === RA-BC 样本加权 (可选) ===
    if rabc_weights_provider is not None:
        rabc_batch_weights, rabc_batch_stats = rabc_weights_provider.compute_batch_weights(batch)

    # === 前向传播 (混合精度) ===
    with accelerator.autocast():
        if rabc_batch_weights is not None:
            # 逐样本损失 + 加权
            per_sample_loss, output_dict = policy.forward(batch, reduction="none")
            epsilon = 1e-6
            loss = (per_sample_loss * rabc_batch_weights).sum() / (rabc_batch_weights.sum() + epsilon)
        else:
            loss, output_dict = policy.forward(batch)

    # === 反向传播 ===
    accelerator.backward(loss)

    # === 梯度裁剪 ===
    if grad_clip_norm > 0:
        grad_norm = accelerator.clip_grad_norm_(policy.parameters(), grad_clip_norm)
    else:
        grad_norm = torch.nn.utils.clip_grad_norm_(policy.parameters(), float("inf"))

    # === 优化器步进 ===
    with lock if lock else nullcontext():
        optimizer.step()
    optimizer.zero_grad()

    # === 学习率调度器步进 ===
    if lr_scheduler is not None:
        lr_scheduler.step()

    # === 策略内部更新 (如 EMA 目标网络) ===
    if has_method(unwrapped_policy, "update"):
        unwrapped_policy.update()

    # === 记录指标 ===
    train_metrics.loss = loss.item()
    train_metrics.grad_norm = grad_norm.item()
    train_metrics.lr = optimizer.param_groups[0]["lr"]
    train_metrics.update_s = time.perf_counter() - start_time
    return train_metrics, output_dict

5.2 EpisodeAwareSampler

用于 Diffusion Policy 训练的专用采样器：

class EpisodeAwareSampler:
    def __init__(self, dataset_from_indices, dataset_to_indices,
                 drop_n_first_frames=0, drop_n_last_frames=0, shuffle=False):
        """
        构建每个 episode 的有效帧范围:
          start = dataset_from_index + drop_n_first_frames
          end   = dataset_to_index - drop_n_last_frames

        drop_n_last_frames 丢弃 episode 末尾被 padding 填充的不完整帧
        """
        self.valid_indices = []
        for ep_start, ep_end in zip(dataset_from_indices, dataset_to_indices):
            valid_start = ep_start + drop_n_first_frames
            valid_end = ep_end - drop_n_last_frames
            if valid_end > valid_start:
                self.valid_indices.extend(range(valid_start, valid_end))

5.3 Delta Timestamps 解析

def resolve_delta_timestamps(cfg, ds_meta):
    """将策略的 delta_indices 转换为时间戳偏移"""
    delta_timestamps = {}
    if cfg.reward_delta_indices:
        delta_timestamps["next.reward"] = [i/ds_meta.fps for i in cfg.reward_delta_indices]
    if cfg.action_delta_indices:
        delta_timestamps["action"] = [i/ds_meta.fps for i in cfg.action_delta_indices]
    # observation_delta_indices 在 dataset.__getitem__ 中自动处理
    return delta_timestamps

6. 处理器管道

6.1 预处理器 (Policy Preprocessor)

输入: batch dict from DataLoader
  │
  ├── RenameObservationsProcessorStep
  │     └── 根据 rename_map 重命名观察键
  │
  ├── [TactileValidationProcessorStep]  (可选)
  │     └── 验证触觉数据形状
  │
  ├── AddBatchDimensionProcessorStep
  │     └── 为标量数据添加 batch 维度
  │
  ├── DeviceProcessorStep
  │     └── 将张量移动到指定设备 (cuda/cpu)
  │
  └── NormalizerProcessorStep
        └── 使用数据集统计量归一化:
            MEAN_STD: (x - mean) / std
            MIN_MAX:  2 * (x - min) / (max - min) - 1
            QUANTILES: (x - q01) / (q99 - q01) * 2 - 1

6.2 后处理器 (Policy Postprocessor)

输入: policy output (normalized actions)
  │
  ├── UnnormalizerProcessorStep
  │     └── 反归一化:
            MEAN_STD: x * std + mean
            MIN_MAX:  (x + 1) / 2 * (max - min) + min
  │
  └── DeviceProcessorStep (CPU)
        └── 移回 CPU

7.2 保存函数

def save_checkpoint(checkpoint_dir, step, cfg, policy, optimizer,
                    scheduler, preprocessor, postprocessor):
    pretrained_dir = checkpoint_dir / "pretrained_model"
    training_dir = checkpoint_dir / "training_state"

    # 1. 保存模型
    policy.save_pretrained(pretrained_dir)
    cfg.save_pretrained(pretrained_dir)  # → train_config.json

    # 2. 保存处理器
    preprocessor.save_pretrained(pretrained_dir)
    postprocessor.save_pretrained(pretrained_dir)

    # 3. 保存训练状态
    save_optimizer_state(optimizer, training_dir)
    save_scheduler_state(scheduler, training_dir)
    save_rng_state(training_dir)
    write_json({"step": step}, training_dir / "training_step.json")

def load_training_state(checkpoint_path, optimizer, lr_scheduler):
    training_dir = checkpoint_path / "training_state"

    step = load_json(training_dir / "training_step.json")["step"]
    load_optimizer_state(optimizer, training_dir)
    load_scheduler_state(lr_scheduler, training_dir)
    load_rng_state(training_dir)

8. RA-BC 加权训练

文件: src/lerobot/utils/rabc.py

8.1 原理

RA-BC（Reward-Aligned Behavior Cloning）使用预计算的 SARM 进度评分来对训练样本进行加权。高质量过渡 → 高权重，低质量过渡 → 低权重。

8.2 训练时集成

# 在 update_policy() 中:
if rabc_weights_provider is not None:
    # 1. 获取逐样本权重
    rabc_batch_weights = rabc_weights_provider.compute_batch_weights(batch)
    # weights 已归一化到 sum = batch_size

    # 2. 获取逐样本损失
    per_sample_loss, output_dict = policy.forward(batch, reduction="none")

    # 3. 加权损失
    # L_RA-BC = Σ(w_i * l_i) / (Σw_i + ε)
    loss = (per_sample_loss * rabc_batch_weights).sum() / (rabc_batch_weights.sum() + epsilon)

8.3 权重计算

class RABCWeights:
    def compute_batch_weights(self, batch):
        # 1. 根据 batch 中的 episode_index + frame_index 查询预计算的进度
        progress_scores = self._lookup_progress(batch)

        # 2. 计算权重
        # delta_i = |progress_i - median_progress|
        # weight_i = 1 / (delta_i + epsilon) if delta_i > kappa else 1.0
        # 归一化: weight_i *= batch_size / sum(weights)
