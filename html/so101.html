LeRobot操作手册

授予接口权限

sudo chmod +666 /dev/ttyACM*——机械臂
sudo chmod +666 /dev/ttyUSB*——传感器

测试传感器

flexitac-find-port
flexitac-heatmap --port /dev/ttyUSB0

遥操作

# 无触觉
lerobot-teleoperate \
--robot.type=so101_follower \
--robot.port=/dev/ttyACM0 \
--robot.id=R07255112 \
--teleop.type=so101_leader \
--teleop.port=/dev/ttyACM1 \
--teleop.id=R07255111
# 有触觉
lerobot-teleoperate \
--robot.type=so_tactile_follower \
--robot.port=/dev/ttyACM0 \
--robot.id=R07255112 \
--teleop.type=so101_leader \
--teleop.port=/dev/ttyACM1 \
--teleop.id=R07255111

录取数据集

# 无触觉

  --robot.type=so101_follower \
  --robot.port=/dev/ttyACM0 \
  --robot.id=R07255112 \
  --robot.cameras='{'handeye': {'type': 'opencv', 'index_or_path': 0, 'width': 640, 'height': 360, 'fps': 30, fourcc: "MJPG"}, 'fixed': {'type': 'opencv', 'index_or_path': 2, 'width': 640, 'height': 360, 'fps': 30, fourcc: "MJPG"}}' \
  --teleop.type=so101_leader \
  --teleop.port=/dev/ttyACM1 \
  --teleop.id=R07255111 \
  --display_data=True \
  --dataset.repo_id=grasp/test1 \
  --dataset.num_episodes=100 \
  --dataset.single_task="Grab the cube from 5 different positions." \
  --dataset.push_to_hub=False \
  --dataset.episode_time_s=20 \
  --dataset.reset_time_s=7
# 有触觉
lerobot-record \
  --robot.type=so_tactile_follower \
  --robot.port=/dev/ttyACM0 \
  --robot.id=R07255112 \
  --robot.cameras='{'front': {'type': 'opencv', 'index_or_path': 0, 'width': 640, 'height': 480, 'target_width': 640, 'target_height': 480, 'fps': 30, fourcc: "MJPG"}, 'side': {'type': 'opencv', 'index_or_path': 2, 'width': 640, 'height': 480,  'target_width': 640, 'target_height': 480, 'fps': 30, fourcc: "MJPG"}}' \
  --robot.tactile_sensors='{primary: {"port": "/dev/ttyUSB0", "baud_rate": 2000000}}' \
  --teleop.type=so101_leader \
  --teleop.port=/dev/ttyACM1 \
  --teleop.id=R07255111 \
  --display_data=True \
  --dataset.repo_id=grasp/tactile1 \
  --dataset.num_episodes=100 \
  --dataset.episode_time_s=30 \
  --dataset.reset_time_s=10 \
  --dataset.single_task="Grab the electronic components from anywhere and place it on the cube which is located at the bla0ck area." \
  --dataset.fps=30 \
  --dataset.push_to_hub=False

  --output_dir=outputs/train/act_test1 \
  --job_name=act_test1 \
  --policy.device=cuda \
  --wandb.enable=false \
  --policy.push_to_hub=false\
