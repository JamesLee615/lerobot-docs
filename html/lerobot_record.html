1. 概述

lerobot_record 是 LeRobot 的数据采集核心。它将遥操作（或策略推断）产生的传感器数据实时记录为结构化的 LeRobot 数据集。

支持的录制模式：

  • 纯遥操作记录：主手控制从手，记录示范数据
  • 策略驱动记录：预训练策略控制机器人，记录评估数据
  • 混合模式：同时配置遥操作和策略进行记录

3. record() 主流程

@parser.wrap()
def record(cfg: RecordConfig) -> LeRobotDataset:
    # === 1. 初始化 ===
    init_logging()
    init_rerun()  # 可选的数据可视化

    # === 2. 创建设备 ===
    robot = make_robot_from_config(cfg.robot)
    teleop = make_teleoperator_from_config(cfg.teleop) if cfg.teleop else None

    # === 3. 创建处理器管道 ===
    # 三个处理器：遥操作动作处理器、机器人动作处理器、机器人观察处理器
    teleop_action_processor, robot_action_processor, robot_observation_processor = \
        make_default_processors()
    # 默认均为 IdentityProcessor (不做任何变换)

    # === 4. 聚合特征定义 ===
    # 将遥操作和机器人观察的特征合并为数据集特征
    dataset_features = combine_feature_dicts(
        aggregate_pipeline_dataset_features(
            pipeline=teleop_action_processor,
            initial_features=create_initial_features(action=robot.action_features),
            use_videos=cfg.dataset.video,
        ),
        aggregate_pipeline_dataset_features(
            pipeline=robot_observation_processor,
            initial_features=create_initial_features(observation=robot.observation_features),
            use_videos=cfg.dataset.video,
        ),
    )

    # === 5. 创建或加载数据集 ===
    if cfg.resume:
        dataset = LeRobotDataset(repo_id, root=root, ...)  # 加载已有
        sanity_check_dataset_robot_compatibility(dataset, robot, ...)
    else:
        sanity_check_dataset_name(repo_id, policy)
        dataset = LeRobotDataset.create(   # 创建新数据集
            repo_id, fps, root=root,
            robot_type=robot.name,
            features=dataset_features,
            use_videos=cfg.dataset.video,
            streaming_encoding=cfg.dataset.streaming_encoding,
            ...
        )

    # === 6. 加载策略 (如果配置) ===
    if cfg.policy is not None:
        policy = make_policy(cfg.policy, ds_meta=dataset.meta)
        preprocessor, postprocessor = make_pre_post_processors(
            policy_cfg=cfg.policy, dataset_stats=dataset.meta.stats, ...
        )

    # === 7. 连接设备 ===
    robot.connect()
    if teleop: teleop.connect()

    # === 8. 触觉传感器元数据记录 ===
    if hasattr(robot, "tactile_sensor_metadata"):
        robot.wait_for_tactile_calibration(timeout_s=...)
        dataset.meta.info["tactile_sensors"] = robot.tactile_sensor_metadata()
        write_info(dataset.meta.info, dataset.meta.root)

    # === 9. 键盘监听 ===
    listener, events = init_keyboard_listener()
    # events = {"exit_early": False, "rerecord_episode": False, "stop_recording": False}
    # 右箭头: exit_early | 左箭头: rerecord_episode | ESC: stop_recording

    # === 10. 录制循环 ===
    with VideoEncodingManager(dataset):
        recorded_episodes = 0
        while recorded_episodes < num_episodes and not events["stop_recording"]:
            # 录制一个 episode
            record_loop(robot, events, fps, ..., control_time_s=episode_time_s, ...)

            # Episode 间重置 (跳过最后一个)
            if not events["stop_recording"] and (recorded_episodes < num_episodes - 1):
                record_loop(robot, events, fps, ..., control_time_s=reset_time_s, dataset=None, ...)

            # 处理重新录制请求
            if events["rerecord_episode"]:
                events["rerecord_episode"] = False
                dataset.clear_episode_buffer()
                continue

            dataset.save_episode()
            recorded_episodes += 1

    # === 11. 收尾 ===
    finally:
        dataset.finalize()            # 关闭 writer, 合并视频, 计算统计量
        robot.disconnect()
        teleop.disconnect() if teleop else None
        if cfg.dataset.push_to_hub:
            dataset.push_to_hub(tags=tags, private=private)

def record_loop(robot, events, fps, ...):
    timestamp = 0
    start_episode_t = time.perf_counter()

    while timestamp < control_time_s:
        start_loop_t = time.perf_counter()

        # 检查是否提前退出
        if events["exit_early"]:
            events["exit_early"] = False
            break

        # 步骤 1: 读取机器人观察
        obs = robot.get_observation()
        # obs = {
        #     "shoulder_pan.pos": 0.52,   # 关节角度
        #     "shoulder_lift.pos": -0.31,
        #     ...,
        #     "front": np.ndarray(H, W, 3),  # 相机图像
        #     "side": np.ndarray(H, W, 3),
        #     "observation.tactile.primary": np.ndarray(12, 32),  # 触觉 (可选)
        # }

        obs_processed = robot_observation_processor(obs)

        # 步骤 2: 构建观察帧
        observation_frame = build_dataset_frame(
            dataset.features, obs_processed, prefix="observation."
        )

        # 步骤 3: 获取动作 (遥操作 或 策略)
        # ... (详见第5节)

        # 步骤 4: 应用机器人动作处理器
        robot_action_to_send = robot_action_processor((action, obs))

        # 步骤 5: 发送动作到机器人
        _sent_action = robot.send_action(robot_action_to_send)

        # 步骤 6: 写入数据集
        if dataset is not None:
            action_frame = build_dataset_frame(dataset.features, action, prefix="action.")
            frame = {**observation_frame, **action_frame, "task": single_task}
            dataset.add_frame(frame)

        # 步骤 7: 可视化 (可选)
        if display_data:
            log_rerun_data(observation=obs_processed, action=action, ...)

        # 步骤 8: 精确等待以维持目标 FPS
        dt_s = time.perf_counter() - start_loop_t
        sleep_time_s = 1 / fps - dt_s
        if sleep_time_s < 0:
            logging.warning(f"Loop running slower ({1/dt_s:.1f} Hz) than target ({fps} Hz)")
        precise_sleep(max(sleep_time_s, 0.0))

        timestamp = time.perf_counter() - start_episode_t

def predict_action(observation, policy, device, preprocessor, postprocessor, ...):
    # 1. 将观察转为 PyTorch 张量
    observation = {k: torch.tensor(v).to(device) for k, v in observation.items()}

    # 2. 通过预处理器
    observation = preprocessor(observation)
    # 步骤: RenameObservations → AddBatchDim → ToDevice → Normalize

    # 3. 策略推理
    with torch.inference_mode(), torch.autocast(device) if use_amp else nullcontext():
        action = policy.select_action(observation)

    # 4. 通过后处理器
    action = postprocessor(action)
    # 步骤: Unnormalize → ToDevice(CPU)

    return action.to("cpu").numpy()

7. 视频编码机制

7.1 三种编码模式

模式 配置 图像写入 视频编码时机 延迟
PNG 模式 streaming_encoding=False 异步写 PNG (AsyncImageWriter) save_episode() 时编码 高
Streaming 模式 streaming_encoding=True 无 (直接送编码器) 实时编码 低
Batched 模式 batchencodingsize > 1 异步写 PNG 积攒多个 episode 后批量编码 最高延迟，最高吞吐

7.2 Streaming 编码流程

Record Loop (每帧):
  camera_frame (HWC uint8)
      │
      ▼
  StreamingVideoEncoder.feed_frame(camera_key, frame, stats_collector)
      │
      ├── 帧进入 per-camera 编码线程队列
      │   _CameraEncoderThread:
      │     while running:
      │       frame = queue.get()
      │       av_frame = av.VideoFrame.from_ndarray(frame, format="rgb24")
      │       encoder.encode(av_frame)  → 写入 mp4 文件
      │       stats_collector.update(frame)  → 在线计算统计量
      │
      ▼
  (非阻塞, 立即返回)

save_episode():
  StreamingVideoEncoder.finish_episode(camera_key)
      │
      ├── 发送 sentinel 停止编码线程
      ├── 等待线程退出
      ├── 收集输出的 mp4 文件路径
      └── 返回每相机统计量

concat (finalize 时):
  将同一 camera key 的多个 episode mp4 合并为一个文件

7.3 支持的编解码器

# libsvtav1 - 默认, 开源 AV1 编码器, 最佳压缩比
